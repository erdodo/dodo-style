# Carousel
## Kullanım
```jsx
import { Carousel } from 'dodo-style'
```

<Carousel indicators={true} responsive={responsiveSettings}>
    {
        images.map((image, index) => (
            <div key={index} className="example1">
                <div style={{ 'backgroundImage': `url(${image})` }}>
                    <span>Slide {index + 1}</span>
                </div>
            </div>
        ))
    }
</Carousel>
