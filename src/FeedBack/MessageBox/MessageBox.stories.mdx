```jsx

```

    title={"Başlık"}
    message={"lorem ipsum dolor sit amet lorem ipsum dolor sit amet lorem ipsum dolor sit amet lorem ipsum dolor sit amet"}
    >
    <Button type={"primary"}>Message Box</Button>
</MessageBox>

## onConfirm - onCancel Örnek

<div style={{display:"flex",flexDirection:"row",alignItems:"center",flexWrap:"wrap"}}>
    <div style={{margin:"5px"}}>
        <MessageBox
            title={"Başlık"}
            message={"lorem ipsum dolor sit amet"}
            onConfirm={()=> alert("Onaylandı")}
            onCancel={()=> alert("Vazgeçildi")}
        >
            <Button type={"primary"}>Message Box</Button>
        </MessageBox>
    </div>
</div>
